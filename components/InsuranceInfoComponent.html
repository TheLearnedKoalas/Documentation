  parentLoadContent(value) {
    if (value === 'Застраховки') {
      this.content = this.getAll();
    } else {
      this.content = [this.insurancesList.find(x => x.name === value)];
    }
  }

  getAll() {
    const all = [];
    this.insurancesList.forEach(x => {
      all.push({ name: x.name, logo: x.logo });
    });
    return all;
  }

  ngOnInit() {
  }
}
